import gspread

from config import GOOGLE_SHEET_ID as SPREADSHEET_ID
import logging
import os
import traceback

# Отключаем прокси
import requests
session = requests.Session()
session.trust_env = False  # Игнорировать системные прокси
os.environ['HTTP_PROXY'] = ''

os.environ['https_proxy'] = ''

logger = logging.getLogger(__name__)

# Настройка доступа к Google Sheets
SCOPES = ['https://www.googleapis.com/auth/spreadsheets']
SERVICE_ACCOUNT_FILE = 'credentials.json'

class GoogleSheetsManager:
    def __init__(self):
        self.creds = None
        self.client = None
        self.spreadsheet = None
        self.connected = False
        
    def connect(self):
        """Подключение к Google Таблицам"""
        try:
            print("🔄 Подключаюсь к Google Таблицам...")
            
            # Проверяем наличие файла с ключами
            if not os.path.exists(SERVICE_ACCOUNT_FILE):
                print(f"❌ Файл {SERVICE_ACCOUNT_FILE} не найден!")
                print("   Создайте файл credentials.json с ключами сервисного аккаунта")
                return False
            
            # Создаем учетные данные
            self.creds = Credentials.from_service_account_file(
                SERVICE_ACCOUNT_FILE, 
                scopes=SCOPES
            )
            
            # Авторизуемся
            self.client = gspread.authorize(self.creds)
            
            # Открываем таблицу
            if not SPREADSHEET_ID or SPREADSHEET_ID == 'ВАШ_GOOGLE_SHEET_ID':
                print("❌ GOOGLE_SHEET_ID не настроен в config.py!")
                return False
            
            self.spreadsheet = self.client.open_by_key(SPREADSHEET_ID)
            
            # Проверяем существование основного листа
            try:
                worksheet = self.spreadsheet.worksheet("Лист1")
                print("✅ Основной лист найден")
            except gspread.exceptions.WorksheetNotFound:
                print("ℹ️ Лист 'Лист1' не найден, будет создан автоматически")
            
            self.connected = True
            print("✅ Подключение к Google Таблицам успешно установлено")
            return True
            
        except Exception as e:
            print(f"❌ Ошибка подключения к Google Таблицам: {e}")
            traceback.print_exc()
            return False

    def is_connected(self):
        """Проверка состояния подключения"""
        return self.connected

    def is_user_registered(self, user_id):
        """Проверка, регистрировался ли пользователь ранее"""
        try:
            if not self.connected:
                return False
            
            sheet = self.get_worksheet("Лист1")
            if not sheet:
                return False
            
            # Получаем все данные
            all_data = sheet.get_all_values()
            
            # Ищем пользователя по ID (колонка H, индекс 7)
            for row in all_data[1:]:  # пропускаем заголовки
                if len(row) > 7 and str(row[7]) == str(user_id):
                    print(f"⚠️ Пользователь {user_id} уже регистрировался")
                    return True
            
            return False
            
        except Exception as e:
            print(f"❌ Ошибка проверки регистрации: {e}")
            return False

    def find_user_row(self, user_id):
        """Найти номер строки пользователя по ID"""
        try:
            if not self.connected:
                return None
            
            sheet = self.get_worksheet("Лист1")
            if not sheet:
                return None
            
            # Получаем все данные
            all_data = sheet.get_all_values()
            
            # Ищем пользователя по ID (колонка H, индекс 7)
            for i, row in enumerate(all_data):
                if i == 0:  # пропускаем заголовки
                    continue
                if len(row) > 7 and str(row[7]) == str(user_id):
                    return i + 1  # +1 потому что гугл считает с 1
            
            return None
            
        except Exception as e:
            print(f"❌ Ошибка поиска пользователя: {e}")
            return None

    def update_user(self, user_data):
        """Обновление данных существующего пользователя (или создание нового)"""
        try:
            print(f"🔄 Обновляю данные пользователя {user_data.get('user_id')}")
            
            if not self.connected:
                print("❌ Нет подключения к Google Таблицам")
                return False
            
            sheet = self.get_worksheet("Лист1")
            if not sheet:
                return False
            
            # Находим строку пользователя
            row_num = self.find_user_row(user_data.get('user_id'))
            
            # Формируем строку данных (12 колонок)
            row_data = [
                datetime.now().strftime("%Y-%m-%d %H:%M:%S"),  # A - Дата регистрации
                user_data.get('full_name', ''),               # B - ФИО
                user_data.get('university', ''),              # C - Уч. заведение
                user_data.get('specialty', ''),               # D - Специальность
                user_data.get('course', ''),                  # E - Курс
                user_data.get('phone', ''),                   # F - Номер телефона
                user_data.get('need_letter', 'Нет'),          # G - Наличие отпроски
                str(user_data.get('user_id', '')),            # H - Telegram user_id
                user_data.get('username', ''),                # I - Имя пользователя
                user_data.get('debate_date', ''),             # J - Дата дебатов
                user_data.get('confirmation', 'Ожидает'),     # K - Статус
                ''                                             # L - Оценка (пусто)
            ]
            
            if row_num:
                # Обновляем существующую строку
                for col_num, value in enumerate(row_data, start=1):
                    sheet.update_cell(row_num, col_num, value)
                print(f"✅ Данные пользователя {user_data.get('user_id')} обновлены (строка {row_num})")
            else:
                # Добавляем новую строку
                sheet.append_row(row_data)
                print(f"✅ Новый пользователь {user_data.get('user_id')} добавлен")
            
            return True
            
        except Exception as e:
            print(f"❌ Ошибка обновления пользователя: {e}")
            traceback.print_exc()
            return False

    def delete_user(self, user_id):
        """Удаление пользователя по ID"""
        try:
            print(f"🔧 Попытка удалить пользователя {user_id}")
            
            if not self.connected:
                print("⚠️ Нет подключения к Google Таблицам")
                return False
            
            sheet = self.get_worksheet("Лист1")
            if not sheet:
                return False
            
            # Получаем все значения
            all_data = sheet.get_all_values()
            
            if len(all_data) <= 1:
                print("ℹ️ Таблица пуста, удалять нечего")
                return False
            
            # Ищем строку с нужным user_id (колонка H, индекс 7)
            row_to_delete = None
            for i, row in enumerate(all_data):
                if i == 0:  # пропускаем заголовки
                    continue
                if len(row) > 7 and str(row[7]) == str(user_id):
                    row_to_delete = i + 1  # +1 потому что гугл считает с 1
                    break
            
            if row_to_delete:
                # Удаляем строку
                sheet.delete_rows(row_to_delete)
                print(f"✅ Пользователь {user_id} удален из таблицы (строка {row_to_delete})")
                return True
            else:
                print(f"ℹ️ Пользователь {user_id} не найден в таблице")
                return False
                
        except Exception as e:
            print(f"❌ Ошибка удаления пользователя: {e}")
            traceback.print_exc()
            return False

    def get_worksheet(self, sheet_name="Лист1"):
        """Получение рабочего листа (создает при необходимости)"""
        try:
            if not self.connected:
                print("⚠️ Нет подключения к Google Таблицам")
                return None
            
            try:
                # Пытаемся получить существующий лист
                worksheet = self.spreadsheet.worksheet(sheet_name)
                return worksheet
                
            except gspread.exceptions.WorksheetNotFound:
                # Создаем новый лист
                print(f"📝 Создаю новый лист '{sheet_name}'...")
                worksheet = self.spreadsheet.add_worksheet(
                    title=sheet_name, 
                    rows=1000, 
                    cols=20
                )
                
                # Если это основной лист для регистрации, добавляем заголовки
                if sheet_name == "Лист1":
                    headers = [
                        'Дата регистрации',          # A
                        'ФИО',                        # B
                        'Уч. заведение',              # C
                        'Специальность',              # D
                        'Курс',                       # E
                        'Номер телефона',             # F
                        'Наличие отпроски',           # G
                        'Telegram user_id пользователя', # H - ИСПРАВЛЕНО
                        'Имя пользователя',           # I
                        'Дата запланированных дебатов', # J
                        'Статус подтверждения участия', # K
                        'Оценка мероприятия'          # L
                    ]
                    worksheet.append_row(headers)
                    print(f"✅ Добавлены заголовки: {headers}")
                
                return worksheet
                
        except Exception as e:
            print(f"❌ Ошибка получения листа '{sheet_name}': {e}")
            return None

    def get_all_users(self):
        """Получение всех пользователей"""
        try:
            if not self.connected:
                print("⚠️ Нет подключения к Google Таблицам")
                return []
        
            sheet = self.get_worksheet("Лист1")
            if not sheet:
                return []
        
            # Получаем все записи
            data = sheet.get_all_records()
            print(f"📋 Получено {len(data)} записей из таблицы")
        
            # Преобразуем данные для удобного доступа
            formatted_data = []
            for record in data:
                formatted_record = {
                    'date': record.get('Дата регистрации', ''),
                    'full_name': record.get('ФИО', ''),
                    'university': record.get('Уч. заведение', ''),
                    'specialty': record.get('Специальность', ''),
                    'course': record.get('Курс', ''),
                    'phone': record.get('Номер телефона', ''),
                    'need_letter': record.get('Наличие отпроски', ''),
                    'user_id': record.get('Telegram user_id пользователя', ''),  # ИСПРАВЛЕНО
                    'username': record.get('Имя пользователя', ''),
                    'debate_date': record.get('Дата запланированных дебатов', ''),
                    'confirmation': record.get('Статус подтверждения участия', ''),
                    'rating': record.get('Оценка мероприятия', '')
                }
                formatted_data.append(formatted_record)
        
            return formatted_data
        
        except Exception as e:
            print(f"❌ Ошибка получения пользователей: {e}")
            traceback.print_exc()
            return []

    def update_confirmation(self, user_id, confirmation_status):
        """Обновление статуса подтверждения в таблице"""
        try:
            if not self.connected:
                return False
            
            sheet = self.get_worksheet("Лист1")
            if not sheet:
                return False
            
            # Получаем все данные
            all_data = sheet.get_all_values()
            
            # Ищем пользователя по ID (колонка H, индекс 7)
            for i, row in enumerate(all_data):
                if i == 0:  # пропускаем заголовки
                    continue
                if len(row) > 7 and str(row[7]) == str(user_id):
                    # Обновляем статус в колонке K (индекс 10)
                    sheet.update_cell(i + 1, 11, confirmation_status)
                    print(f"✅ Подтверждение для {user_id} обновлено на '{confirmation_status}'")
                    return True
            
            print(f"❌ Пользователь {user_id} не найден")
            return False
            
        except Exception as e:
            print(f"❌ Ошибка обновления подтверждения: {e}")
            return False

    def save_feedback(self, user_id, feedback_text, rating):
        """Сохранение обратной связи"""
        try:
            if not self.connected:
                return False
            
            # Сначала обновляем оценку в основном листе
            sheet = self.get_worksheet("Лист1")
            if sheet:
                all_data = sheet.get_all_values()
                for i, row in enumerate(all_data):
                    if i == 0:
                        continue
                    if len(row) > 7 and str(row[7]) == str(user_id):
                        # Обновляем оценку в колонке L (индекс 11)
                        sheet.update_cell(i + 1, 12, str(rating))
                        print(f"✅ Оценка {rating} для {user_id} сохранена")
                        break
            
            # Также сохраняем в лист обратной связи
            feedback_sheet = self.get_worksheet("Обратная связь")
            if not feedback_sheet:
                return True  # Уже обновили в основном листе
            
            # Проверяем заголовки
            headers = feedback_sheet.row_values(1)
            if not headers or len(headers) < 3:
                new_headers = [
                    'Дата', 'ID пользователя', 'Текст отзыва', 
                    'Оценка', 'Статус'
                ]
                try:
                    feedback_sheet.clear()
                    feedback_sheet.append_row(new_headers)
                except:
                    pass
            
            # Формируем строку
            timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            row = [
                timestamp,
                str(user_id),
                feedback_text,
                str(rating),
                'Новый'
            ]
            
            try:
                # Сохраняем
                feedback_sheet.append_row(row)
                print(f"✅ Обратная связь от {user_id} сохранена")
            except:
                pass
            
            return True
            
        except Exception as e:
            print(f"❌ Ошибка сохранения обратной связи: {e}")
            return False

    def save_survey(self, user_id, answers):
        """Сохранение результатов опроса"""
        try:
            if not self.connected:
                return False
            
            # Получаем или создаем лист для опросов
            sheet = self.get_worksheet("Опросы")
            if not sheet:
                return False
            
            # Проверяем заголовки
            headers = sheet.row_values(1)
            if not headers or len(headers) < 3:
                new_headers = ['Дата', 'ID пользователя', 'Ответы']
                try:
                    sheet.clear()
                    sheet.append_row(new_headers)
                except:
                    pass
            
            # Формируем строку
            timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            # Преобразуем список ответов в строку
            if isinstance(answers, list):
                answers_str = " | ".join([str(a) for a in answers])
            else:
                answers_str = str(answers)
            
            row = [timestamp, str(user_id), answers_str]
            
            try:
                # Сохраняем
                sheet.append_row(row)
                print(f"✅ Опрос от {user_id} сохранен")
            except:
                pass
            
            return True
            
        except Exception as e:
            print(f"❌ Ошибка сохранения опроса: {e}")
            return False

    def get_pending_confirmations(self):
        """Получение списка ожидающих подтверждения"""
        try:
            if not self.connected:
                return []
            
            users = self.get_all_users()
            pending = []
            
            for user in users:
                if user.get('confirmation') == 'Ожидает':
                    pending.append(user)
            
            print(f"📊 Найдено {len(pending)} ожидающих подтверждения")
            return pending
            
        except Exception as e:
            print(f"❌ Ошибка получения ожидающих подтверждения: {e}")
            return []
